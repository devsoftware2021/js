Basic File I/O (Optional)

1. Browser-based File Operations

a. Using FileReader in the Browser

The FileReader API allows you to read the contents of a file chosen by the user.

    
    <input type="file" id="fileInput" />
    <pre id="fileContent"></pre>

    <script>
        document.getElementById('fileInput').addEventListener('change', function(event) {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                
                reader.onload = function(e) {
                    document.getElementById('fileContent').textContent = e.target.result;
                };

                reader.onerror = function(e) {
                    console.error("Error reading file:", e);
                };

                reader.readAsText(file);
            }
        });
    </script>

b. Using Blob in the Browser

The Blob object represents binary data, which can be used for storing and handling large chunks of data.

    
    const data = new Blob(["Hello, world!"], { type: "text/plain" });
    const link = document.createElement('a');
    link.href = URL.createObjectURL(data);
    link.download = "hello.txt"; // File name
    link.click();

2. Node.js File Operations

a. Reading a File in Node.js

In Node.js, you can use the fs.readFile() method to read the contents of a file asynchronously.

    
    const fs = require('fs');

    fs.readFile('example.txt', 'utf8', (err, data) => {
        if (err) {
            console.error("Error reading file:", err);
            return;
        }
        console.log("File content:", data);
    });

b. Writing a File in Node.js

You can write data to a file using fs.writeFile().

    
    const fs = require('fs');

    const content = "Hello, this is a test file!";

    fs.writeFile('output.txt', content, (err) => {
        if (err) {
            console.error("Error writing file:", err);
            return;
        }
        console.log("File written successfully.");
    });
